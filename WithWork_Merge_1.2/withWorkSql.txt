--  DDL for View TODO_GUEST_INFO
--------------------------------------------------------

  CREATE OR REPLACE FORCE VIEW "WITHWORK"."TODO_GUEST_INFO" ("T_NUM", "EDIT_NUM", "ORDER_M", "ORDERER", "DEPART", "O_ROLE", "WORKER_M", "WORKER", "W_DEPART", "W_ROLE", "P_NUM", "P_NAME", "D_NUM", "TITLE", "TEXT", "ORDERDATE", "ENDDATE", "BETWEENDATE", "PROGRESS", "FINALDATE") AS 
  SELECT T2.T_NUM, T2.EDIT_NUM, T2.ORDER_M, T2.G_NAME ORDERER, T2.DEPART, T2.ROLE O_ROLE, T2.WORKER_M, G2.G_NAME WORKER, D2.DEPART W_DEPART,
       R2.ROLE W_ROLE, T2.P_NUM, T2.NAME P_NAME, T2.D_NUM, T2.TITLE, T2.TEXT, T2.ORDERDATE, T2.ENDDATE, T2.BETWEENDATE, T2.PROGRESS, T2.FINALDATE 
        FROM (SELECT T.T_NUM, T.EDIT_NUM, T.ORDER_M, R.ROLE, D.DEPART, G.G_NAME, T.WORKER_M, T.P_NUM, P.NAME, T.D_NUM, T.TITLE, T.TEXT, 
        TO_CHAR(T.ORDERDATE, 'YYYY-MM-DD') ORDERDATE, TO_CHAR(T.ENDDATE, 'YYYY-MM-DD') ENDDATE, 
        T.ENDDATE-T.ORDERDATE BETWEENDATE, T.PROGRESS, TO_CHAR(T.FINALDATE, 'YYYY-MM-DD') FINALDATE 
        FROM GUEST G, MEMBER M, ROLES R,DEPARTMENT D, TODO T, PROJECT P 
        WHERE G.G_NUM = M.G_NUM AND M.R_NUM = R.R_NUM AND M.M_NUM = T.ORDER_M AND T.P_NUM = P.P_NUM AND D.D_NUM = M.D_NUM) T2, MEMBER M2, ROLES R2, DEPARTMENT D2, GUEST G2
        WHERE G2.G_NUM = M2.G_NUM AND M2.R_NUM = R2.R_NUM AND M2.M_NUM = T2.WORKER_M AND M2.D_NUM = D2.D_NUM ORDER BY EDIT_NUM DESC
;
